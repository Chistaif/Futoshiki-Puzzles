\subsection{Iterative Deepening Search (IDS)}
  \subsubsection{Idea}
  IDS combines DFS's low memory footprint with BFS's optimality under unit costs. It runs depth-limited DFS repeatedly with limits $L=0,1,2,\dots$
  until a goal is found. To mitigate repeated-state loops within a depth-limited run, we use a \emph{path-based} visited set (cycle checking). A global
  visited set is typically \emph{not} used across iterations because it can incorrectly prune a state reachable at a smaller depth in a later
  iteration.

  \subsubsection{Pseudocode}
 % --- Phần 1: Hàm IDS chính ---
\begin{algorithm}[H]
    \caption{Iterative Deepening Search (IDS)}
    \begin{algorithmic}[1]
        \Require $s_0, \textit{max\_depth}, \textit{max\_nodes}, \textit{max\_time}$
        \Ensure $(solved, moves, \textit{expanded})$
        
        \Function{IDS\_Search}{$s_0, \textit{max\_depth}, \textit{max\_nodes}, \textit{max\_time}$}
            \State $\textit{expanded} \gets 0$; \quad $\textit{begin} \gets \text{Now}()$
            \State $\textit{best\_prog} \gets \langle \text{ProgressScore}(s_0), \emptyset \rangle$
            
            \For{$L \gets 1$ \textbf{to} $\textit{max\_depth}$}
                \If{$\textit{expanded} \ge \textit{max\_nodes}$ \textbf{or} $\text{Now}() - \textit{begin} \ge \textit{max\_time}$}
                    \State \textbf{break}
                \EndIf
                \State $\textit{path} \gets [\ ]$; \quad $\textit{visited} \gets \text{Map}(\text{Key}(s_0) \mapsto 0)$
                \State $\textit{result} \gets \Call{DLS}{s_0, \textit{path}, \textit{visited}, L, \textit{expanded}, \textit{best\_prog}}$
                \If{$\textit{result} = \textbf{true}$}
                    \State \Return $(\textbf{true}, \textit{path}, \textit{expanded})$
                \EndIf
                \If{$\textit{result} = \textbf{false}$} \State \textbf{break} \EndIf
            \EndFor
            \State \Return $(\textbf{false}, \textit{best\_prog.path}, \textit{expanded})$
        \EndFunction
    \end{algorithmic}
\end{algorithm}

% --- Phần 2: Hàm DLS bổ trợ ---
\begin{algorithm}[H]
    \caption{Depth-Limited Search (DLS) helper}
    \begin{algorithmic}[1]
        \Function{DLS}{$s, \textit{path}, \textit{visited}, L, \textit{expanded}, \textit{best\_prog}$}
            \If{$\textit{expanded} \ge \textit{max\_nodes}$ \textbf{or} $\text{Now}() - \textit{begin} \ge \textit{max\_time}$}
                \State \Return \textbf{false}
            \EndIf
            \If{$\text{Goal}(s)$} \State \Return \textbf{true} \EndIf
            \If{$\text{Length}(\textit{path}) \ge L$} \State \Return \text{CUTOFF} \EndIf
            
            \State $\textit{expanded} \gets \textit{expanded} + 1$
            \State $\textit{successors} \gets \text{GenerateMoves}(s)$ \Comment{Sorted by Foundation}
            
            \State $\textit{cutoff} \gets \textbf{false}$
            \ForAll{$\langle move, s' \rangle \in \textit{successors}$}
                \State $d' \gets \text{Length}(\textit{path}) + 1$
                \If{$\textit{visited}.get(\text{Key}(s'), \infty) \le d'$} \State \textbf{continue} \EndIf
                
                \State $\textit{visited}[\text{Key}(s')] \gets d'$; \quad $\textit{path.Append}(move)$
                \State $\textit{res} \gets \Call{DLS}{s', \textit{path}, \textit{visited}, L, \textit{expanded}, \textit{best\_prog}}$
                
                \If{$\textit{res} = \textbf{true}$} \State \Return \textbf{true} \EndIf
                \If{$\textit{res} = \text{CUTOFF}$} \State $\textit{cutoff} \gets \textbf{true}$ \EndIf
                \State $\textit{path.Pop}()$
            \EndFor
            \State \Return $(\textit{cutoff} \text{ ? } \text{CUTOFF} : \textbf{false})$
        \EndFunction
    \end{algorithmic}
\end{algorithm}

  \subsubsection{Algorithm Analysis}
  \begin{itemize}
    \item \textbf{Time:} $O(b^d)$ (with a constant-factor overhead from re-expanding upper levels).
    \item \textbf{Space:} $O(bd)$ (DFS stack plus on-path set), typically far smaller than BFS.
    \item \textbf{Completeness:} Complete for finite $b$ (eventually reaches limit $L=d$).
    \item \textbf{Optimality (unit-cost):} Optimal because the first goal discovered occurs at minimum depth $d$, and $c(n,a,n')=1$.
  \end{itemize}
