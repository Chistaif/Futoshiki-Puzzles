\subsection{A* Search}
  \subsubsection{Idea}
  A* prioritizes nodes by
  \[
  f(n) = g(n) + h(n),
  \]
  where $g(n)$ is the cost-so-far and $h(n)$ estimates the remaining cost to reach the goal. For Freecell we use the \emph{Foundation Gap} heuristic:
  \[
  h_{\text{gap}}(n) = 52 - \sum_{s \in \{C,D,H,S\}} F_s,
  \]
  i.e., the number of cards not yet placed into foundations. We maintain an open priority queue keyed by $(f,g)$ and an explored-cost table $
  \textsc{bestG}$ to prune dominated repeats.
\subsubsection{Heuristic Analysis and Properties}
The implementation defaults to the \textbf{Foundation Gap} heuristic ($h_{\text{gap}}$).

\paragraph{1. Default Heuristic Analysis ($h_{\text{gap}}$)}
Defined as $h_{\text{gap}}(n) = 52 - \sum F_s$, representing the cards remaining outside foundations.
\begin{itemize}
    \item \textbf{Admissibility:} Each card requires at least one move to reach a foundation pile. Thus, $h_{\text{gap}}(n)$ never exceeds the actual cost to the goal, ensuring optimality.
    \item \textbf{Consistency:} A single legal move changes the foundation count by at most 1. Therefore, $h(n) \le c(n, a, n') + h(n')$ holds for unit cost $c=1$.
\end{itemize}

\paragraph{2. Blocking Heuristic ($h_{\text{block}}$)}
To provide a more informed estimate, the \textit{Blocking Heuristic} extends $h_{\text{gap}}$ by accounting for physical constraints in the cascades:
\[ h_{\text{block}}(n) = h_{\text{gap}}(n) + \sum_{\text{suit}} \text{blockers}(\text{suit}) \]
A \textbf{blocker} is defined as any card positioned on top of a "target" card (the immediate next rank needed for a foundation pile). For each suit, the algorithm locates the target card in the cascades and counts how many cards must be moved before that target becomes accessible. This heuristic remains admissible because those blocking cards \textit{must} be moved at least once to clear the path to the foundation.

\paragraph{3. Comparative Analysis: $h_{\text{gap}}$ vs. $h_{\text{block}}$}
The solver also provides the \textbf{Blocking Heuristic} ($h_{\text{block}}$) as an alternative.
\begin{table}[H]
    \centering
    \small
    \begin{tabular}{|l|p{5cm}|p{5cm}|}
    \hline
    \textbf{Heuristic} & \textbf{Pros} & \textbf{Cons} \\ \hline
    \textbf{Foundation Gap} & Fast calculation; guaranteed shortest path with weight $w=1.0$. & Less informed; ignores card accessibility buried in cascades. \\ \hline
    \textbf{Blocking} & Highly informed; prunes large search spaces by penalizing "buried" cards. & Computational overhead per node due to cascade scanning. \\ \hline
    \end{tabular}
    \caption{Trade-offs between the available heuristics.}
\end{table}
\paragraph{4. State Normalization and Tie-breaking}
\begin{itemize}
    \item \textbf{Canonical Representation:} To prevent redundant expansions, states are normalized via a \texttt{SearchKey} that sorts cascades and free cells, ensuring logically identical boards are treated as one.
    \item \textbf{Tie-breaking:} When $f(n)$ values are equal, the priority queue selects nodes with a lower $h(n)$ and a higher \textit{Progress Score} (calculated based on foundations and open slots) to drive the search toward the goal faster.
\end{itemize}

  \subsubsection{Pseudocode}
 \begin{algorithm}[H]
    \caption{$A^*$ Heuristics and State Normalization}
    \begin{algorithmic}[1]
        \Function{FoundationGap}{$s$}
            \State \Return $52 - \sum s.\text{foundations}$ \Comment{Admissible heuristic}
        \EndFunction
        
        \Statex
        \Function{BlockingHeuristic}{$s$}
            \State $\mathit{h} \gets \Call{FoundationGap}{s}$
            \State $\mathit{blockers} \gets 0$
            \ForAll{$\text{suit} \in \{C, D, H, S\}$}
                \State $\mathit{target} \gets s.\text{foundations}[\text{suit}] + 1$
                \State Find position of $\mathit{target}$ in cascades
                \State $\mathit{blockers} \gets \mathit{blockers} + \text{cards on top of target}$
            \EndForAll
            \State \Return $\mathit{h} + \mathit{blockers}$
        \EndFunction

        \Statex
        \Function{SearchKey}{$s$}
            \State $\mathit{can\_free} \gets \text{sort}(s.\text{free\_cells})$
            \State $\mathit{can\_casc} \gets \text{sort}(\text{tuple}(c) \text{ for } c \text{ in } s.\text{cascades})$
            \State \Return $\langle \mathit{can\_casc}, \mathit{can\_free}, s.\text{foundations} \rangle$ \Comment{Canonical form}
        \EndFunction
    \end{algorithmic}
\end{algorithm}


\begin{algorithm}[H]
    \caption{$A^*$ Session Initialization}
    \begin{algorithmic}[1]
        \State $\mathit{start} \gets \text{Clone}(s_0) \text{ with } g=0$
        \State $\mathit{start}.h \gets \Call{Heuristic}{\mathit{start}}$
        \State $\mathit{frontier} \gets \text{PriorityQueue}(\langle f, h, \mathit{tie}, \mathit{start} \rangle) \text{ where } f = g + w \cdot h$
        \State $\mathit{best\_cost} \gets \text{Map}(\Call{SearchKey}{\mathit{start}} \mapsto 0)$
        \State $\mathit{best\_prog} \gets \langle \text{Foundations}, \text{OpenSlots}, -h, -g \rangle$ \Comment{Tie-breaking metrics}
    \end{algorithmic}
\end{algorithm}


\begin{algorithm}[H]
    \caption{$A^*$ Main Advance Loop}
    \begin{algorithmic}[1]
        \Function{AStar\_Advance}{$\mathit{max\_nodes}, \mathit{max\_time}$}
            \While{$\mathit{frontier} \neq \emptyset$}
                \State \textbf{Check} $\mathit{max\_nodes}$ \textbf{and} $\mathit{max\_time}$ \textbf{limits}
                \State $(\_, \_, \_, \mathit{current}) \gets \mathit{frontier}.\text{PopMin}()$
                \State $\mathit{ckey} \gets \Call{SearchKey}{\mathit{current}}$
                
                \If{$\mathit{current}.g > \mathit{best\_cost}.\text{get}(\mathit{ckey}, \infty)$} \textbf{continue} \EndIf
                
                \If{$\text{Goal}(\mathit{current})$} \Return $(\textbf{true}, \mathit{current})$ \EndIf
                
                \State $\mathit{expanded} \gets \mathit{expanded} + 1$
                \State \Call{UpdateBestProgress}{\mathit{current}}
                
                \State $\mathit{moves} \gets \text{Sort}(\text{LegalMoves}(\mathit{current})) \text{ by greedy priority}$
                \ForAll{$m \in \mathit{moves}$}
                    \State $s' \gets \text{Apply}(\mathit{current}, m)$; \quad $s'.h \gets \Call{Heuristic}{s'}$
                    \State $\mathit{nkey} \gets \Call{SearchKey}{s'}$
                    
                    \If{$s'.g < \mathit{best\_cost}.\text{get}(\mathit{nkey}, \infty)$}
                        \State $\mathit{best\_cost}[\mathit{nkey}] \gets s'.g$
                        \State $\mathit{priority} \gets s'.g + w \cdot s'.h$
                        \State $\mathit{frontier.Push}(\langle \mathit{priority}, s'.h, \mathit{tie}\text{++}, s' \rangle)$
                    \EndIf
                \EndFor
            \EndWhile
            \State \Return $(\textbf{false}, \mathit{best\_state})$ \Comment{Fallback to best progress}
        \EndFunction
    \end{algorithmic}
\end{algorithm}


\begin{algorithm}[H]
    \caption{$A^*$ Progress Evaluation}
    \begin{algorithmic}[1]
        \Function{ProgressScore}{$s$}
            \State $\mathit{found} \gets \sum s.\text{foundations}$
            \State $\mathit{slots} \gets \text{CountEmptyFreeCells}(s) + \text{CountEmptyCascades}(s)$
            \State \Return $\langle \mathit{found}, \mathit{slots}, -s.h, -s.g \rangle$ \Comment{Lexicographical comparison}
        \EndFunction
    \end{algorithmic}
\end{algorithm}

  \subsubsection{Algorithm Analysis}
  \begin{itemize}
    \item \textbf{Time:} Exponential in the worst case (often expressed as $O(b^d)$), but typically much smaller than uninformed search when $h$ is
  informative.
    \item \textbf{Space:} $O(V)$ for $\textsc{bestG}$ and the open list; in practice can be the limiting factor on large Freecell instances.
    \item \textbf{Completeness:} Complete for finite $b$ and positive step costs (here $1$).
    \item \textbf{Optimality (unit-cost):} With admissible $h$ (and in particular with consistent $h_{\text{gap}}$), A* returns an optimal minimum-move.
  solution.
  \end{itemize}
